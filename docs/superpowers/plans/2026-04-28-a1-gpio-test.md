```cpp
namespace {

bool parse_u32(const std::string& value, uint32_t* out) {
    char* end = nullptr;
    unsigned long parsed = std::strtoul(value.c_str(), &end, 10);
    if (!end || *end != '\0') return false;
    *out = static_cast<uint32_t>(parsed);
    return true;
}

bool parse_pin(const std::string& value, int* pin) {
    char* end = nullptr;
    long parsed = std::strtol(value.c_str(), &end, 10);
    if (!end || *end != '\0') return false;
    if (parsed != 0 && parsed != 2 && parsed != 8 && parsed != 9 && parsed != 10) return false;
    *pin = static_cast<int>(parsed);
    return true;
}

bool parse_mode(const std::string& value, GpioTestMode* mode) {
    if (value == "tx") {
        *mode = GpioTestMode::Tx;
        return true;
    }
    if (value == "rx") {
        *mode = GpioTestMode::Rx;
        return true;
    }
    if (value == "loop") {
        *mode = GpioTestMode::Loop;
        return true;
    }
    return false;
}

}  // namespace
```

- [ ] **Step 2: Implement exact parser behavior**

Add `ParseGpioTestArgs(...)`:

```cpp
bool ParseGpioTestArgs(const std::vector<std::string>& args,
                       GpioTestConfig* config,
                       std::string* error_message) {
    if (!config || !error_message) return false;

    *config = GpioTestConfig{};
    bool have_pin = false;
    bool have_mode = false;

    for (size_t i = 0; i < args.size(); ++i) {
        const std::string& arg = args[i];
        if (arg == "--pin") {
            if (i + 1 >= args.size() || !parse_pin(args[++i], &config->pin)) {
                *error_message = "invalid --pin, expected one of 0/2/8/9/10; usage: --gpio-test --pin 8 --mode tx [--period-ms 500] [--duration-s 10]";
                return false;
            }
            have_pin = true;
        } else if (arg == "--mode") {
            if (i + 1 >= args.size() || !parse_mode(args[++i], &config->mode)) {
                *error_message = "invalid --mode, expected tx|rx|loop; usage: --gpio-test --pin 8 --mode tx [--period-ms 500] [--duration-s 10]";
                return false;
            }
            have_mode = true;
        } else if (arg == "--period-ms") {
            if (i + 1 >= args.size() || !parse_u32(args[++i], &config->period_ms) || config->period_ms == 0) {
                *error_message = "invalid --period-ms, expected positive integer";
                return false;
            }
        } else if (arg == "--duration-s") {
            if (i + 1 >= args.size() || !parse_u32(args[++i], &config->duration_s) || config->duration_s == 0) {
                *error_message = "invalid --duration-s, expected positive integer";
                return false;
            }
        } else {
            *error_message = std::string("unknown argument: ") + arg;
            return false;
        }
    }

    if (!have_pin) {
        *error_message = "missing --pin; usage: --gpio-test --pin 8 --mode tx [--period-ms 500] [--duration-s 10]";
        return false;
    }
    if (!have_mode) {
        *error_message = "missing --mode; usage: --gpio-test --pin 8 --mode tx [--period-ms 500] [--duration-s 10]";
        return false;
    }
    if (config->mode == GpioTestMode::Rx) {
        config->period_ms = 500;
    }
    return true;
}
```

```cpp
bool setup_gpio(const GpioTestConfig& config, gpio_mode_t mode, GpioRuntime* runtime) {

        log_error("gpio_set_mode", config.pin);
        return false;
    }
    return true;
}

void cleanup_gpio(GpioRuntime* runtime) {
```

```cpp
int run_tx(const GpioTestConfig& config) {
    GpioRuntime runtime;
    if (!setup_gpio(config, GPIO_MODE_OUTPUT, &runtime)) return 1;

    const uint32_t iterations = (config.duration_s * 1000U) / config.period_ms;
    int level = 0;
    for (uint32_t i = 0; i < iterations; ++i) {
        level = (i % 2 == 0) ? 1 : 0;
        if (gpio_write_pin(runtime.handle,
                           runtime.pin_mask,
                           level ? GPIO_PIN_SET : GPIO_PIN_RESET) != 0) {
            log_error("gpio_write_pin", config.pin);
            cleanup_gpio(&runtime);
            return 1;
        }
        log_line("tx", config.pin, "set", level);
        usleep(config.period_ms * 1000U);

    return 0;
```

```cpp
int run_rx(const GpioTestConfig& config) {
    GpioRuntime runtime;
    if (!setup_gpio(config, GPIO_MODE_INPUT, &runtime)) return 1;

    const uint64_t deadline_ms = monotonic_time_ms() + static_cast<uint64_t>(config.duration_s) * 1000ULL;
    uint16_t state = 0;
    int last_level = -1;
    uint32_t change_count = 0;

    while (monotonic_time_ms() < deadline_ms) {
        if (gpio_read_pin(runtime.handle, runtime.pin_mask, &state) != 0) {
            log_error("gpio_read_pin", config.pin);
            cleanup_gpio(&runtime);
            return 1;
        }
        const int level = (state & runtime.pin_mask) ? 1 : 0;
        if (level != last_level) {
            log_line("rx", config.pin, "change", level);
            last_level = level;
            change_count += 1;
        }
        usleep(10000);
    }

    std::fprintf(stdout, "[gpio_test] mode=rx pin=%d action=summary changes=%u\n", config.pin, change_count);
    std::fflush(stdout);
    cleanup_gpio(&runtime);
    return 0;
}
```
